<S> ::= (<S>) | <F> | <G> | <H> | <I> | <J> | ε

<A> ::= var_id                                  // VARIÁVEIS
<B> ::= int_num                                 // INTEIROS
<C> ::= + | - | * | /                           // OPERADORES

<D> ::= <A> | <B> | <A><C><D> | <B><C><D>       // EXPRESSÃO
<E> ::= <D> == <D> | <D> != <D>                 // CONDIÇÃO

<F> ::= var <A> = <D>;                          // DECLARAÇÃO
<G> ::= <A> = <D>;                              // ATRIBUIÇÃO

<H> ::= <I>(<E>) {<S>}                          // BLOCO CONDICIONAL
<I> ::= se | enquanto                           // ESTRUTURA CONDICIONAL

<J> ::= imprime (<D>) ;                         // FUNÇÃO
